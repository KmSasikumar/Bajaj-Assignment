SQL Problem Solutions

Question 1 (Odd Registration)

WITH DeptStats AS (
    SELECT d.department_name, AVG(e.salary) as avg_salary
    FROM departments d
    JOIN employees e ON d.department_id = e.department_id
    GROUP BY d.department_id, d.department_name
    HAVING COUNT(e.employee_id) >= 5
)
SELECT department_name, avg_salary
FROM DeptStats
WHERE avg_salary = (SELECT MAX(avg_salary) FROM DeptStats);

Question 2 (Even Registration)

WITH RankedEmployees AS (
    SELECT 
        d.department_name,
        e.first_name || ' ' || e.last_name as full_name,
        e.salary,
        DENSE_RANK() OVER (PARTITION BY d.department_id ORDER BY e.salary DESC) as rank
    FROM employees e
    JOIN departments d ON e.department_id = d.department_id
)
SELECT department_name, full_name, salary
FROM RankedEmployees
WHERE rank <= 3
ORDER BY department_name, salary DESC;
